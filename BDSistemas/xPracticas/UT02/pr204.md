```python
def find_by_radius(lat, lon, distance=2000):
    resultados = r.geosearch(
        name="poi:locations",
        latitude=lat,
        longitude=lon,
        radius=distance,
        unit="m"
    )

    return resultados
```

```python
def find_nearest(lat, lon):
    punto = r.geosearch(
        name="poi:locations", 
        latitude=lat, 
        longitude=lon,
        radius=10000000000000000000,
        count=1,
        withdist=True
    )
    id, dist = punto[0]
    nombre = r.hget("poi:info", id)
    lugar = {
        "loc": nombre,
        "dist": dist
    }
    return lugar

usuario = {
    "lat": 40.41677,
    "lon": -3.70379
}

loc = find_nearest(usuario["lat"], usuario["lon"])
print("El POI mas cercano es ",loc["loc"], "que esta a ",loc["dist"],"km")
```
